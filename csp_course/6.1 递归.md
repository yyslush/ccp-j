# 从两个程序体会递归的工作流程
### 程序一

    #include<iostream>
    using namespace std;

    int n,mul=1;

    void dfs(int n){

        for(int i=n;i>=1;--i)
            mul*=i;

    }

    int main(){

        cin>>n;
        dfs(n);
        cout<<mul;
        return 0;

    }

### 程序二


    #include<iostream>
    using namespace std;

    int n,mul=1;

    void dfs(int n){
        if(n==1)
            return ; 
    
        mul*=n;
        dfs(n-1);
    }

    int main(){

        cin>>n;
        dfs(n);
        cout<<mul;
        
        return 0;

    }
